import pandas as pd

# Supondo que você tenha um DataFrame chamado df com os dados
df['data_venda'] = pd.to_datetime(df['data_venda'])  # Certifique-se de que a data está no formato correto
df['trimestre'] = df['data_venda'].dt.to_period('Q')  # Cria a coluna de trimestre

# Agrupa por trimestre e produto, somando a quantidade vendida
produtos_trimestre = df.groupby(['trimestre', 'produto'])['quantidade_vendida'].sum().reset_index()

# Ordena os produtos por quantidade vendida
produtos_trimestre_ordenados = produtos_trimestre.sort_values(by='quantidade_vendida', ascending=False)

# Exibe os 5 produtos mais vendidos de cada trimestre
for trimestre in produtos_trimestre_ordenados['trimestre'].unique():
    print(f"Produtos mais vendidos no {trimestre}:")
    top_produtos = produtos_trimestre_ordenados[produtos_trimestre_ordenados['trimestre'] == trimestre].head(5)
    print(top_produtos[['produto', 'quantidade_vendida']])
